    },
    bicos: {
      titulo: "Limpeza de Bicos",
      texto: "Limpeza de bicos injetores. Valor médio: R$100",
    },
    escapamento: {
      titulo: "Escapamentos",
      texto: "Manutenção do sistema de escapamento. Valor médio: R$120",
    },
    freios: {
      titulo: "Freios / ABS",
      texto: "Sistema de freios e ABS. Segurança total. Valor médio: R$150",
    },
    embreagem: {
      titulo: "Embreagem",
      texto: "Troca de embreagem. Valor médio: R$250",
    },
    radiador: {
      titulo: "Radiadores",
      texto: "Sistema de arrefecimento. Valor médio: R$180",
    },
    suspensao: {
      titulo: "Suspensão",
      texto: "Conforto e estabilidade. Valor médio: R$200",
    },
    revisao: {
      titulo: "Revisão Completa",
      texto: "Revisão geral do veículo. Valor médio: R$150",
    },
    inspecao: {
      titulo: "Diagnóstico / Inspeção",
      texto: "Diagnóstico completo do veículo. Valor médio: R$80",
    },
  };

  const modal = document.getElementById("modal");
  const modalTitle = document.getElementById("modalTitle");
  const modalText = document.getElementById("modalText");

  document.querySelectorAll(".card").forEach((card) => {
    card.addEventListener("click", () => {
      const key = card.getAttribute("data-key");
      const dados = dadosServicos[key];
      if (dados) {
        modalTitle.textContent = dados.titulo;
        modalText.textContent = dados.texto;
        modal.classList.add("show");
      }
    });
  });

  window.addEventListener("click", (event) => {
    if (event.target === modal) {
      modal.classList.remove("show");
    }
  });

  function atualizarPreco() {
    let valor = document.getElementById("servico").value;
    if (!valor) return;
    let preco = valor.split("|")[1];
    document.getElementById("preco").innerText = "Valor médio: R$" + preco;
  }

  function enviarWhats(event) {
    event.preventDefault();

    let nome = document.getElementById("nome").value;
    let telefone = document.getElementById("telefone").value;
    let partes = document.getElementById("servico").value.split("|");
    let servico = partes[0];
    let preco = partes[1];
    let data = document.getElementById("data").value;
    let hora = document.getElementById("hora").value;

    let numero = "5599999999999"; // Coloque seu número com DDD aqui

    let mensagem = `Agendamento:%0A
Nome: ${nome}%0A
Telefone: ${telefone}%0A
Serviço: ${servico}%0A
Valor: R$${preco}%0A
Data: ${data}%0A
Hora: ${hora}`;

    window.open(`https://wa.me/${numero}?text=${mensagem}`, "_blank");
  }
